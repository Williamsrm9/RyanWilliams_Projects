    class Car
    {
        public Car() { }


        public Car(string carModel)
        {
            alloyWheels = false;
            spoilers = false;
            Tintedwindows = false;
            premiumPackage = false;
            conveniencePackage = false;
            techPackage = false;
            lightingPackage = false;
            luxeryPackage = false;
            carName = "default";
            CarModel = carModel;
            TotalPrice = 0;


        }
        public string carName { get; set; }
        public bool alloyWheels { get; set; }
        public bool spoilers { get; set; }
        public bool Tintedwindows { get; set; }
        public bool premiumPackage { get; set; }
        public bool conveniencePackage { get; set; }


        public bool techPackage { get; set; }
        public bool lightingPackage { get; set; }


        public bool luxeryPackage { get; set; }


        public string CarModel { get; set; }
        public double TotalPrice { get; set; }


        /*
            this is the options method it accpets a car object and displays a series of options
            this method calls the print method
        */
        public void Options(Car car)
        {
            string temp;
            bool flag = true;
            Console.WriteLine("Would you like to add options? Enter Y for yes or N for no: ");
            temp = Console.ReadLine();
            if (temp == "y" || temp == "Y")
            {
                while (flag)
                {
                    Console.WriteLine("Press 1 to add a spoiler, 2 for tinted windows, 3 for alloy wheels, 4 for Premium Package, 5 for Convenience Package, 6 for Tech Package, 7 for Lighting Package, 8 for Luxury Package, 9 to exit the ADD FEATURES menu: ");
                    temp = Console.ReadLine();
                    int a = int.Parse(temp);




                    switch (a)
                    {
                        case 1: //option to add a spoiler
                            car.spoilers = true;
                            AddToPrice(2000);
                            break;
                        case 2: //option to add tinted windows
                            car.Tintedwindows = true;
                            AddToPrice(1000);
                            break;
                        case 3: //option to add alloy wheels
                            car.alloyWheels = true;
                            AddToPrice(3000);
                            break;
                        case 4: //option to add premium package
                            car.premiumPackage = true;
                            AddToPrice(3000);
                            break;
                        case 5: //option to add convenience package
                            car.conveniencePackage = true;
                            AddToPrice(3000);
                            break;
                        case 6: //option to add tech package
                            car.techPackage = true;
                            AddToPrice(2000);
                            break;
                        case 7: //option to add lighting package
                            car.lightingPackage = true;
                            AddToPrice(1500);
                            break;
                        case 8: //option to add luxury package
                            car.luxeryPackage = true;
                            AddToPrice(3000);
                            break;
                        case 9: //exit switch statement, move to displaying output


                            printOutput(car);
                            flag = false;
                            break;
                        default:
                            break;
                    }




                }


            }


            if (temp == "n" || temp == "N") //User has selected not to add any extra features
            {
                Console.WriteLine("No extra features have been added."); //Notify user that no features have been added.
                Console.WriteLine(" ");
                printOutput(car); //Display order output


            }
        }


        /*
            this is the print out put method it accpes a car object and prints out all of the
            options to the console along with the total price
        */
        public void printOutput(Car car)
        {
            Console.WriteLine("-------REVIEW ORDER-------");
            Console.WriteLine(car.carName); //display name of the car
            Console.WriteLine("Car Type: " + CarModel); //display the type of the car
            Console.WriteLine(" ");
            Console.WriteLine("Added Features"); //indicate features will now be displayed
            if (car.spoilers == true)
            {
                Console.WriteLine("Spoiler: Yes"); //indicate this feature was chosen
            }
            else
            {
                Console.WriteLine("Spoiler: No"); //indicate this feature was not chosen
            }


            if (car.Tintedwindows == true)
            {
                Console.WriteLine("Tinted Windows: Yes"); //indicate this feature was chosen
            }
            else
            {
                Console.WriteLine("Tinted Windows: No"); //indicate this feature was not chosen
            }


            if (car.alloyWheels == true)
            {
                Console.WriteLine("Alloy Wheels: Yes"); //indicate this feature was chosen
            }
            else
            {
                Console.WriteLine("Alloy Wheels: No"); //indicate this feature was not chosen
            }


            if (car.premiumPackage == true)
            {
                Console.WriteLine("Premium Package: Yes"); //indicate this feature was chosen
            }
            else
            {
                Console.WriteLine("Premium Package: No"); //indicate this feature was not chosen
            }


            if (car.conveniencePackage == true)
            {
                Console.WriteLine("Convenience Package: Yes"); //indicate this feature was chosen
            }
            else
            {
                Console.WriteLine("Convenience Package: No"); //indicate this feature was not chosen
            }
            if (car.techPackage == true)
            {
                Console.WriteLine("Tech Package: Yes"); //indicate this feature was chosen
            }
            else
            {
                Console.WriteLine("Tech Package: No"); //indicate this feature was not chosen
            }
            if (car.lightingPackage == true)
            {
                Console.WriteLine("Lighting Package: Yes"); //indicate this feature was chosen
            }
            else
            {
                Console.WriteLine("Lighting Package: No"); //indicate this feature was not chosen
            }
            if (car.luxeryPackage == true)
            {
                Console.WriteLine("Luxury Package: Yes"); //indicate this feature was chosen
            }
            else
            {
                Console.WriteLine("Luxury Package: No"); //indicate this feature was not chosen
            }
            Console.WriteLine(" ");
            Console.WriteLine("TOTAL PRICE: $" + TotalPrice); //display total price
        }
        // this is the add to price method it accepts a double and returns the total price of the
        // car and options
        public double AddToPrice(double currentBalance)
        {
            TotalPrice = TotalPrice + currentBalance;
            Console.WriteLine("The price as configured is: $" + TotalPrice);
            return TotalPrice;


        }
